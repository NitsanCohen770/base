import { colorPalette } from './color-palette';

## React Component for rendering text

A basic div that renders some text

### Colors Available:

```js
<div className={colorPalette.color}>
  <div style={{ backgroundColor: 'var(--text-gray)' }}> Gray</div>
</div>
```

### Using props to customize the text

Modify the text to see it change live:

```js live
<ColorPalette text="hello from ColorPalette" />
```
